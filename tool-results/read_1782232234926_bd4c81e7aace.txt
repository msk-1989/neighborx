     1→// NeighborX — Hyperlocal Community & Commerce Super App
     2→// PostgreSQL-backed (Neon serverless). Uses String for enum-like fields
     3→// (no native PG enums) and no arrays — keeps the schema portable.
     4→//
     5→// Both local dev and Vercel production use PostgreSQL (Neon). A single
     6→// DATABASE_URL (Neon pooled) + DIRECT_DATABASE_URL (Neon direct) drive
     7→// everything, so the generated Prisma client is always a postgres client —
     8→// no provider swapping needed.
     9→
    10→generator client {
    11→  provider = "prisma-client-js"
    12→}
    13→
    14→datasource db {
    15→  provider  = "postgresql"
    16→  url       = env("DATABASE_URL")
    17→  directUrl = env("DIRECT_DATABASE_URL")
    18→}

   404→// IAM / RBAC — Identity & Access Management

   406→// Pillar: IAM (Role-Based Access Control). Every screen, button, API and
   407→// action is controlled by Roles + Permissions. A user can hold multiple
   408→// roles; each role grants a set of permissions.
   409→
   410→// All possible roles (user-facing + 10-level admin hierarchy)
   411→model Role {
   412→  id          String   @id @default(cuid())
   413→  // code: GUEST | RESIDENT | VERIFIED_RESIDENT | BUSINESS_OWNER |
   414→  //       SERVICE_PROVIDER | EMPLOYER | PROPERTY_OWNER | VOLUNTEER |
   415→  //       SOCIETY_ADMIN | AREA_MODERATOR | CITY_MODERATOR | BUSINESS_ADMIN |
   416→  //       ORG_ADMIN | SUPPORT_ADMIN | OPERATIONS_ADMIN | REVENUE_ADMIN |
   417→  //       COMPLIANCE_ADMIN | SUPER_ADMIN
   418→  code        String   @unique
   419→  name        String
   420→  description String
   421→  // level: 0=user roles, 1-10=admin hierarchy (Society Admin=1 ... Super Admin=10)
   422→  level       Int      @default(0)
   423→  // category: USER | ADMIN
   424→  category    String   @default("USER")
   425→  isSystem    Boolean  @default(true) // system roles cannot be deleted
   426→  permissions RolePermission[]
   427→  users       UserRole[]
   428→  createdAt   DateTime @default(now())
   429→}
   430→
   431→model Permission {
   432→  id          String   @id @default(cuid())
   433→  // code: e.g. CREATE_POST, DELETE_POST, VERIFY_USER, MANAGE_CITY,
   434→  //       APPROVE_BUSINESS, VIEW_REVENUE, MANAGE_USERS, etc.
   435→  code        String   @unique
   436→  name        String
   437→  description String
   438→  // module: which feature area this permission belongs to
   439→  // COMMUNITY | MARKETPLACE | BUSINESS | SERVICE | JOB | PROPERTY |
   440→  // SAFETY | SOCIETY | CIVIC | NEWS | REWARDS | AI | FINANCE | CMS |
   441→  // SETTINGS | TRUST | IAM | SUPPORT | OPERATIONS | COMPLIANCE
   442→  module      String   @default("COMMUNITY")
   443→  // action: CREATE | READ | UPDATE | DELETE | APPROVE | VERIFY | MANAGE | VIEW
   444→  action      String   @default("READ")
   445→  roles       RolePermission[]
   446→  createdAt   DateTime @default(now())
   447→}
   448→
