     1→// NeighborX — Hyperlocal Community & Commerce Super App
     2→// PostgreSQL-backed (Neon serverless). Uses String for enum-like fields
     3→// (no native PG enums) and no arrays — keeps the schema portable.
     4→//
     5→// Both local dev and Vercel production use PostgreSQL (Neon). A single
     6→// DATABASE_URL (Neon pooled) + DIRECT_DATABASE_URL (Neon direct) drive
     7→// everything, so the generated Prisma client is always a postgres client —
     8→// no provider swapping needed.
     9→
    10→generator client {
    11→  provider = "prisma-client-js"
    12→}
    13→
    14→datasource db {
    15→  provider  = "postgresql"
    16→  url       = env("DATABASE_URL")
    17→  directUrl = env("DIRECT_DATABASE_URL")
    18→}

   398→// IAM / RBAC — Identity & Access Management
   399→// =====================================================================
   400→// Pillar: IAM (Role-Based Access Control). Every screen, button, API and
   401→// action is controlled by Roles + Permissions. A user can hold multiple
   402→// roles; each role grants a set of permissions.
   403→
   404→// All possible roles (user-facing + 10-level admin hierarchy)
   405→model Role {
   406→  id          String   @id @default(cuid())
   407→  // code: GUEST | RESIDENT | VERIFIED_RESIDENT | BUSINESS_OWNER |
   408→  //       SERVICE_PROVIDER | EMPLOYER | PROPERTY_OWNER | VOLUNTEER |
   409→  //       SOCIETY_ADMIN | AREA_MODERATOR | CITY_MODERATOR | BUSINESS_ADMIN |
   410→  //       ORG_ADMIN | SUPPORT_ADMIN | OPERATIONS_ADMIN | REVENUE_ADMIN |
   411→  //       COMPLIANCE_ADMIN | SUPER_ADMIN
   412→  code        String   @unique
   413→  name        String
   414→  description String
   415→  // level: 0=user roles, 1-10=admin hierarchy (Society Admin=1 ... Super Admin=10)
   416→  level       Int      @default(0)
   417→  // category: USER | ADMIN
   418→  category    String   @default("USER")
   419→  isSystem    Boolean  @default(true) // system roles cannot be deleted
   420→  permissions RolePermission[]
   421→  users       UserRole[]
   422→  createdAt   DateTime @default(now())
   423→}
   424→
   425→model Permission {
   426→  id          String   @id @default(cuid())
   427→  // code: e.g. CREATE_POST, DELETE_POST, VERIFY_USER, MANAGE_CITY,
   428→  //       APPROVE_BUSINESS, VIEW_REVENUE, MANAGE_USERS, etc.
   429→  code        String   @unique
   430→  name        String
   431→  description String
   432→  // module: which feature area this permission belongs to
   433→  // COMMUNITY | MARKETPLACE | BUSINESS | SERVICE | JOB | PROPERTY |
   434→  // SAFETY | SOCIETY | CIVIC | NEWS | REWARDS | AI | FINANCE | CMS |
   435→  // SETTINGS | TRUST | IAM | SUPPORT | OPERATIONS | COMPLIANCE
   436→  module      String   @default("COMMUNITY")
   437→  // action: CREATE | READ | UPDATE | DELETE | APPROVE | VERIFY | MANAGE | VIEW
   438→  action      String   @default("READ")
   439→  roles       RolePermission[]
   440→  createdAt   DateTime @default(now())
   441→}
   442→
   443→// Many-to-many: Role <-> Permission
   444→model RolePermission {
   445→  id           String     @id @default(cuid())
   446→  roleId       String
   447→  role         Role       @relation(fields: [roleId], references: [id], onDelete: Cascade)
   448→  permissionId String
   449→  permission   Permission @relation(fields: [permissionId], references: [id], onDelete: Cascade)
   450→  createdAt    DateTime   @default(now())
   451→
   452→  @@unique([roleId, permissionId])
